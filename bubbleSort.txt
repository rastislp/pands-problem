def bubbleSort(nlist):

    for passnum in range(len(nlist)-1,0,-1):

        for i in range(passnum):

            if nlist[i]>nlist[i+1]:

                temp = nlist[i]

                nlist[i] = nlist[i+1]

                nlist[i+1] = temp

 

import random

 

 

 

 

nlist = [random.randint(0, 1000) for x in range(1, 11)]

print(nlist)

bubbleSort(nlist)

print(nlist)
